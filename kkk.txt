#include <stdio.h>

int main(void)
{
	int x, y;

	for (x = 0; x < 7; x++) {
		for (y = 0; y <= x; y++)
			printf("*");
		printf("\n");

	}
	return 0;
}

#include <stdio.h>

int main(void)
{
	int x, y;

	for (x = 6; x > 0; x--) {
		for (y = 6; y >= x; y--)
			printf("*");
		for (y = 0; y < x; y++)
			printf("%d", y + 1);
		printf("\n");

	}
	return 0;
}
